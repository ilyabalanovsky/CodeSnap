import { generateColors } from '@codesnap/core';

/**
 * Generates an aurora borealis inspired gradient pattern
 * @param intensity The brightness of the colors (0-1)
 */
export async function createAuroraPattern(intensity: number = 0.8) {
  const basePalette = await generateColors({
    theme: 'northern-lights',
    contrast: 'high'
  });
  
  return `linear-gradient(135deg, ${basePalette.join(', ')})`;
}
